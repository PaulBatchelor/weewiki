int weewiki_open(weewiki_d *ww, const char *filename)
{
    sqlite3 *db;
    int rc;

    ww->db = NULL;
    rc = sqlite3_open(filename, &db);
    if (rc) {
        fprintf(stderr,
                "Could not open database: %s",
                sqlite3_errmsg(db));
        sqlite3_close(db);
        return 0;
    }

    ww->db = db;
    weewiki_create_tables(ww);
    return 1;
}

static int p_push(int argc, char *argv[])
{
    weewiki_d ww;
    FILE *fp;
    char *buf;
    size_t sz;
    sqlite3 *db;
    sqlite3_stmt *stmt;
    struct stat statbuf;
    time_t mtime;
    int rc;

    if (argc < 3) {
        fprintf(stderr,
                "Usage: %s file key\n",
                argv[0]);
        return 1;
    }

    weewiki_init(&ww);
    weewiki_open(&ww, "a.db");
    fp = fopen(argv[1], "r");

    if (fp == NULL) {
        fprintf(stderr,
                "Could not open file %s\n",
                argv[1]);
        return 1;
    }

    fseek(fp, 0, SEEK_END);
    sz = ftell(fp);
    buf = calloc(1, sz + 1);
    fseek(fp, 0, SEEK_SET);
    fread(buf, 1, sz, fp);
    fclose(fp);
    stat(argv[1], &statbuf);

    mtime = statbuf.st_mtime;

    db = ww.db;

    sqlite3_prepare_v2(db,
                       "INSERT OR REPLACE INTO wiki"
                       "(key, value, mtime)\n"
                       "VALUES(?1,?2,?3);",
                       -1,
                       &stmt,
                       NULL);

    sqlite3_bind_text(stmt, 1, argv[2], -1, NULL);
    sqlite3_bind_text(stmt, 2, buf, sz, NULL);
    sqlite3_bind_int(stmt, 3, mtime);

    rc = sqlite3_step(stmt);

    if (rc != SQLITE_DONE) {
        fprintf(stderr,
                "SQLite error: %s\n",
                sqlite3_errmsg(db));
    }
    sqlite3_finalize(stmt);

    free(buf);
    weewiki_close(&ww);
    weewiki_clean(&ww);
    return 0;
}

static int p_pull(int argc, char *argv[])
{
    weewiki_d ww;
    FILE *fp;
    size_t sz;
    sqlite3 *db;
    sqlite3_stmt *stmt;
    int rc;

    if (argc < 3) {
        fprintf(stderr,
                "Usage: %s file key\n",
                argv[0]);
        return 1;
    }

    weewiki_init(&ww);
    weewiki_open(&ww, "a.db");


    db = ww.db;

    sqlite3_prepare_v2(db,
                       "SELECT value FROM wiki WHERE(key==?1);",
                       -1,
                       &stmt,
                       NULL);
    sqlite3_bind_text(stmt, 1, argv[1], -1, NULL);

    rc = sqlite3_step(stmt);

    if (rc != SQLITE_ROW) {
        fprintf(stderr,
                "Could not find key '%s'\n",
                argv[1]);
        sqlite3_finalize(stmt);
        return 1;
    }


    fp = fopen(argv[2], "w");

    if (fp == NULL) {
        fprintf(stderr,
                "Could not open file %s\n",
                argv[1]);
        return 1;
    }
    sz = sqlite3_column_bytes(stmt, 0);
    fwrite(sqlite3_column_text(stmt, 0), 1, sz, fp);
    fclose(fp);
    sqlite3_finalize(stmt);
    weewiki_close(&ww);
    weewiki_clean(&ww);
    return 0;
}
